                 rewardNano := int64(dbTx.ExecutorReward * 1e9)
                 
                 // –î–æ–ø—É—Å–∫–∞–µ–º –Ω–µ–±–æ–ª—å—à—É—é –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç—å –∏–ª–∏ —Ç–æ—á–Ω–æ–µ —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ
                 if valNano == rewardNano {
                     correctAmount = true
                 }
                 break
             }
        }
        
        if foundRecipient && correctAmount {
             log.Printf("PaymentTracker: Verified recipient and amount for task %s", dbTx.TaskID)
             pt.markTransactionConfirmed(...)
             break
        } else {
             log.Printf("PaymentTracker: Mismatch in recipient/amount for QueryID match! Potential collision or attack.")
        }
    }
}
